TPROXY маршрутизация
# nftables правила (упрощённо)
nft add rule inet 4eburnet prerouting
meta l4proto { tcp, udp } tproxy to 127.0.0.1:7893
# маршрутизация
ip rule add fwmark 0x1 table 100
ip route add local 0.0.0.0/0 dev lo table 100
